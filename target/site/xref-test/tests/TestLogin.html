17  public class TestLogin {
18      private WebDriver driver;
19      private Login login;
20  
21      @Before
22      public void SetUp() {
23          driver = new FirefoxDriver();
24          login = new Login(driver);
25          driver.manage().timeouts().implicitlyWait(15, TimeUnit.SECONDS);
26          driver.manage().timeouts().pageLoadTimeout(60, TimeUnit.SECONDS);
27      }
28  
29      @Category({Critical.class})
30      @Test
31      public void succesfullLogin() {
32          login.with("seleniummz@gmail.com","seleniummz12345");
33          assertTrue("element is not found",
34                  login.verifyLoginPage());
35          login.logout();
36          assertTrue("Not Logged Out",
37                  login.verifyLoggedOut());
38      }
39  
40      @Category({Major.class})
41      @Test
42      public void succesfullSendAndReceive() {
43          login.with("seleniummz@gmail.com","seleniummz12345");
44          assertTrue("element is not found",
45                  login.verifyLoginPage());
46          login.composeEmail("seleniummz@gmail.com","succesfullSendAndReceive Test Case","Any Text");
47          login.openEmail();
48          assertEquals("No Subject", "succesfullSendAndReceive Test Case",
49                  driver.findElement(By.cssSelector("h2[class='hP']")).getText());
50          assertEquals("No Body Text", "Any Text",
51                  driver.findElement(By.cssSelector("div[class='adn ads'] div[dir='ltr']")).getText());
52          login.deleteEmail();
53          login.logout();
54  
55  
56      }
57  
58      @After
59      public void tearDown() {
60          driver.quit();
61      }
